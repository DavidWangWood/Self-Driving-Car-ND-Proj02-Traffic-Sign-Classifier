Step 1: Dataset Summary & Exploration¶

The pickled data is a dictionary with 4 key/value pairs:

  • 'features' is a 4D array containing raw pixel data of the traffic sign images, (num examples, width, height, channels).

def regularization_cost (weights, regularization):
    reg_cost = 0
    for weight in weights:
        # https://www.tensorflow.org/api_docs/python/nn/losses#l2_loss
        reg_cost += regularization * tf.nn.l2_loss(weight)
    return reg_cost

def LeNet_2(x):
    # x is INPUT
    with tf.name_scope('conv_1'):
        conv1_w = tf.Variable(tf.truncated_normal(shape=(5, 5, 3, 6), stddev = 0.1))
        conv1_b = tf.Variable(tf.zeros(6))
        conv1 = tf.nn.conv2d(x, conv1_w, strides=[1, 1, 1, 1], padding='VALID') + conv1_b
        # Do RELU here
        conv1 = tf.nn.tanh(conv1)
        # Add maxpool 
        conv1 = tf.nn.max_pool(conv1, ksize=[1,2,2,1], strides=[1,2,2,1], padding ='VALID')
    
    #second Conv (twice color channel)
    with tf.name_scope('conv_2'):
        conv2_w = tf.Variable(tf.truncated_normal(shape=(5, 5, 6, 16), stddev = 0.1))
        conv2_b = tf.Variable(tf.zeros(16))
        conv2 = tf.nn.conv2d(conv1, conv2_w, strides=[1, 1, 1, 1], padding='VALID') + conv2_b
        # Do RELU here
        conv1 = tf.nn.tanh(conv2)
        # Add maxpool 
        conv2 = tf.nn.max_pool(conv2, ksize=[1,2,2,1], strides=[1,2,2,1], padding ='VALID')
        
    # Leveraging from LeNet I do changed and increased the size of zeros and color channel. Think like we scan over through the 
    # pixels inside the second layer.
    
    # Let's connect the First layer
    with tf.name_scope('fc_1'):
        fc1 = flatten(conv2)
        fc1_shape = (fc1.get_shape().as_list()[-1],1024)
        fc1_w = tf.Variable(tf.truncated_normal(shape=(fc1_shape),stddev=0.1))
        fc1_b = tf.Variable(tf.zeros(1024))
        fc1 = tf.matmul(fc1,fc1_w)+fc1_b
        fc1 = tf.nn.tanh(fc1)
        fc1 = tf.nn.dropout(fc1, keep_prob)
    
    # Let's connect second layer
    with tf.name_scope('fc_2'):
        fc2_w = tf.Variable(tf.truncated_normal(shape=(1024,1024),stddev=0.1))
        fc2_b = tf.Variable(tf.zeros(1024))
        fc2 = tf.matmul(fc1,fc2_w)+fc2_b
        fc2 = tf.nn.tanh(fc2)
        fc2 = tf.nn.dropout(fc2,keep_prob)
    
    #Do logits here
    with tf.name_scope('logit'):
        logits_w = tf.Variable(tf.truncated_normal(shape=(1024, number_class), stddev=0.1))
        logits_b = tf.Variable(tf.zeros(number_class))
        return (tf.matmul(fc2,logits_w)+ logits_b),[conv1_w,conv2_w, fc1_w, fc2_w, logits_w]
